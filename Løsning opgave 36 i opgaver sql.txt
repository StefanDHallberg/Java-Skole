WITH	
gennemsnit			
AS (
		SELECT	stdnr, avg(kar) AS snit
		FROM		studfag
		GROUP BY	stdnr
	)
,		
maxsnit			
AS		
    (
		SELECT	MAX( snit )	AS	max_snit
		FROM	gennemsnit
	)
